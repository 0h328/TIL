```python
def fibo(N):
    if N < 3:  # 3보다 작은 경우 계산 필요 X
        return 1
        
    DP = [1, 1] + [0] * (N-2)  # 점화식을 위한 리스트(초기 1~2는 값으로 지정)

    for i in range(2, N):
        DP[i] = DP[i-2] + DP[i-1]  # 점화식
    return DP[N-1]  # 인덱스로 인해 -1값 반환
```
